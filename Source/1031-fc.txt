int main()
{ 
	char s[100];
	scanf("%s", &s);
	int count = 0;
	for (;s[count]!='\0'; count++)
	{
	}
	int N = count + 2;
	int b, l;
	if ((N%3)==1)
	{
		N = (N / 3 + 1) * 3;
		b = N / 3;
		l = (count - b) / 2;
	}
	else if ((N%3)==2)
	{
		N = N -2;
		N = (N / 3 + 1) * 3;
		b = N / 3;
		l = (count - b) / 2;
		b += 1;
	}
	else
	{
		b = N / 3;
		l = (count - b) / 2;
	}
	//lÐÐ µ×ÐÐ b¸ö
	for (int i = 0; i < l; i++)
	{
		printf("%c", s[i]);
		for (int j = 0; j < b-2; j++)
		{
			printf(" ");
		}
		printf("%c\n", s[count - 1 - i]);
	}
	for (int i =0; i < b; i++)
	{
		printf("%c", s[l + i]);
	}
	return 0;
}
