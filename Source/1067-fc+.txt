#include <iostream>
#include<cstdio>
using namespace std;

int main()
{
    freopen("data.txt","r",stdin);
    int length;
    cin>>length;
    int *d=new int[length];
    int *pos=new int[length];

    for(int i=0;i<length;i++){
        cin>>d[i];
        pos[d[i]]=i;
    }
    int sum=0;
    for(int i=0;i<length;i++){
        while(pos[i]!=i){
            while(pos[0]!=0){
                swap(pos[0],pos[pos[0]]);
                sum++;
            }
            if(pos[i]!=i){
                swap(pos[0],pos[i]);
                sum++;
            }
        }
    }
    printf("%d",sum);
    return 0;
}
