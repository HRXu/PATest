void add(vector<int> &d) {
	int length = d.size();
	int *reverse = new int[length];
	for (int i = length-1,j=0; i >-1; i--,j++)
	{
		reverse[j] = d[i];
	}
	int de = 0;//½øÎ»
	for (int i = 0; i < length; i++)
	{
		int tmp = d[i] + reverse[i] + de;
		de=tmp/10;
		d[i] = tmp % 10;
	}
	if (de!=0)
	{
		d.push_back(de);
	}
	free(reverse);
}
bool IsP(vector<int> &d) {
	int length = d.size();
	int c = length / 2;
	for (int i = 0; i < c; i++)
	{
		if (d[i]!=d[length-i-1])
		{
			return false;
		}
	}
	return true;
}
